Formate a tabela do enunciado, de modo que o js entenda. Atente-se ao "enunciado_blocos", ele só deve ser preenchido caso a questão tenha tabelas. Caso não tenha, use "enunciado".


function renderizarEnunciado(questao) {

    // 1. Prioridade: Blocos Estruturados

    if (questao.enunciado_blocos && Array.isArray(questao.enunciado_blocos)) {

        let html = '';

        questao.enunciado_blocos.forEach(bloco => {

            if (bloco.type === 'p') html += `<p>${bloco.content}</p>`;

            else if (bloco.type === 'table') {

                html += '<div class="table-wrapper"><table class="enunciado-table">';

                if (bloco.headerRows) {

                    html += '<thead>';

                    bloco.headerRows.forEach(r => {

                        html += '<tr>';

                        r.forEach(c => html += `<th${c.colspan ? ` colspan="${c.colspan}"` : ''}${c.rowspan ? ` rowspan="${c.rowspan}"` : ''}>${c.content}</th>`);

                        html += '</tr>';

                    });

                    html += '</thead>';

                }

                if (bloco.bodyRows) {

                    html += '<tbody>';

                    bloco.bodyRows.forEach(r => {

                        html += '<tr>';

                        r.forEach(c => html += `<td>${c}</td>`);

                        html += '</tr>';

                    });

                    html += '</tbody>';

                }

                html += '</table></div>';

            }

        });

        return `<div class="enunciado">${html}</div>`;

    }


    // 2. Fallback HTML/Texto

    if (questao.enunciado) {

        const pareceHTML = /<[a-z][\s\S]*>/i.test(questao.enunciado);

        if (pareceHTML) {

            let htmlFinal = questao.enunciado;

            if (htmlFinal.includes('<table') && !htmlFinal.includes('table-wrapper')) {

                htmlFinal = htmlFinal.replace(/<table/g, '<div class="table-wrapper"><table').replace(/<\/table>/g, '</table></div>');

            }

            return `<div class="enunciado enunciado-html">${htmlFinal}</div>`;

        } else {

            return `<div class="enunciado" style="white-space: pre-wrap;">${questao.enunciado}</div>`;

        }

    }

    return '<p class="enunciado" style="color:red;">Erro: Enunciado não encontrado.</p>';

}
